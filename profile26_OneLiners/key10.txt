VAR $k = 0
OLED_CLEAR
OLED_CURSOR 0 0
OLED_PRINT foo    bar    baz
OLED_UPDATE

SWC_FILL 0 0 0
SWC_SET 1 128 0 0
SWC_SET 2 128 0 0
SWC_SET 3 128 0 0

BCLR
WHILE $k == 0
    $k = $_READKEY
    DELAY 50
END_WHILE

IF $k == 1 THEN
  STRINGLN foo
ELSE IF $k == 2 THEN
  STRINGLN bar
ELSE IF $k == 3 THEN
  STRINGLN baz
END_IF

SWC_RESET 99
