DEFAULTCHARDELAY 15

VAR $k = 0
OLED_CLEAR
OLED_CURSOR 0 0
OLED_PRINT MD5   SHA256   SHA1
OLED_CURSOR 0 12
OLED_PRINT Any key to exit
OLED_UPDATE

SWC_FILL 0 0 0
SWC_SET 1 128 0 0
SWC_SET 2 128 0 0
SWC_SET 3 128 0 0

STRING certutil -hashfile 
CTRL SHIFT v
STRING  

BCLR
WHILE $k == 0
    $k = $_READKEY
    DELAY 50
END_WHILE

IF $k == 1 THEN
  STRING MD5 
ELSE IF $k == 2 THEN
  STRING SHA256 
ELSE IF $k == 3 THEN
  STRING SHA256 
END_IF

SWC_RESET 99
